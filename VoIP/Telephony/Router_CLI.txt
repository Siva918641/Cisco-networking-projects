ex
ex

!DHCP

en
config t
ip dhcp pool VoIP
network 192.168.20.0 255.255.255.0
default-router 192.168.20.1
option 150 ip 192.168.20.1
ex
ex
ex

!enabling telephony service

telephony-service
max-dn 6
max-ephones 6
ip source-address 192.168.20.1 port 2000
auto assign 1 to 5
